\nomenclature{Scan}{a procedure which measures the entire area of coverage of the system.}
\nomenclature{Measurement}{a point in the measurement space where something is detected}
\nomenclature{Score}{a measure of the goodness of a measurement-to-track association}
\nomenclature{Dummy measurement}{a self created measurement at the estimated position (with score 0)}
\nomenclature{Measurement list}{a set of measurement which originate from the same scan}
\nomenclature{Target}{an actual object which the system is trying to track}
\nomenclature{Track}{a list of measurement indices, one from each scan, which is believed to originate from the same target}
\nomenclature{Gate}{an area in which a track expects and approves new measurement to associate with itself}
\nomenclature{Track hypothesis}{a new measurement which is inside the gate for an existing track}
